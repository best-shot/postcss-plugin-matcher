## Installation

```bash
npm install postcss-plugin-matcher --save-dev
```

## Usage

```diff
 // example: postcss.config.cjs
 module.exports = {
   plugins: {
-    'tailwindcss': {},
+    'postcss-plugin-matcher': {
+      plugins: ['tailwindcss'],
+      include: ['src/**'],
+      exclude: ['app.css']
+    }
   }
 };
```
